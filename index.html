Closures and Retaining Paths

DOM Elements are GC Roots

        (function() {
          var button = document.getElementById('button'),
              dog = new Dog();

          button.addEventListener('click', function() {
            dog.bark();
          });
        }());

Closures Subtleties

        (function() {
          var dog = new Dog(),
              cat = new Cat();

          button.addEventListener('click', function() {
            dog.bark();
          });

          document.addEventListener('online', function() {
            cat.wakeUp();
          });
        }());
